/**
 * This demonstrates how to execute a compiled .tflite model
 * and access custom parameters/settings appended to the .tflite.
 *
 * Refer to the corresponding README for more details.
 */
#include <cstdio>

// Include the SDK headers
#include "sl_system_init.h"

// This is an optional header
// that demonstrates how to manually configure the system clocks
#include "ml/platform/ml_clock_helper.h"

// Include your compiled model
// Change "image_classification" to your model's name (if necessary)
#include "keyword_spotting_generated.h"
// Also include the custom model parameters #defines
#include "image_classification_generated.parameters.h"

// To keep things simple,
// we globally use the "npu_toolkit" C++ namespace.
// This way, we do not have to manually use it
// for every call.
using namespace npu_toolkit;


/*
 This prints all the parameters that are available as #defines.
 These #defines are generated by the compiler and stored in the
 header file: image_classification_generated.parameters.h
*/
void print_precompiled_defines()
{
  const uint8_t blob[] = SL_TFLITE_MODEL_BLOB;
  const char* class_labels[] = SL_TFLITE_MODEL_CLASS_LABELS;
  const float float_list[] = SL_TFLITE_MODEL_FLOAT_LIST;
  const int int_list[] = SL_TFLITE_MODEL_INT_LIST;
  const char* msg = SL_TFLITE_MODEL_MSG;
  bool normalized_samples = SL_TFLITE_MODEL_NORMALIZE_SAMPLES;
  float period_ms = SL_TFLITE_MODEL_PERIOD_MS;
  int sample_rate = SL_TFLITE_MODEL_SAMPLE_RATE;

  printf("The following parameters where automatically generated \n");
  printf(" into #defines:\n");

  printf("Binary data, key=blob, value={ ");
  for(int i = 0; i < 3; ++i) printf("%02X, ", blob[i]);
  printf(" }\n");

  printf("String list, key=class_labels, value={ ");
  for(int i = 0; i < 5; ++i) printf("%s, ", class_labels[i]);
  printf(" }\n");

  printf("Float list, key=float_list, value={ ");
  for(int i = 0; i < 4; ++i) printf("%3.1f, ", float_list[i]);
  printf(" }\n");

  printf("Integer list, key=int_list, value={ ");
  for(int i = 0; i < 5; ++i) printf("%d, ", int_list[i]);
  printf(" }\n");

  printf("String, key=msg, value=%s\n", SL_TFLITE_MODEL_MSG);
  printf("bool, key=normalized_samples, value=%d\n", SL_TFLITE_MODEL_NORMALIZE_SAMPLES);
  printf("float, key=period_ms, value=%f\n", SL_TFLITE_MODEL_PERIOD_MS);
  printf("int, key=sample_rate, value=%d\n", SL_TFLITE_MODEL_SAMPLE_RATE);
}



int main()
{
  // Initialize the system
  sl_system_init();

  // This is optional,
  // but may be used to configure the platform's max clock rate
  ml_configure_clocks_to_max_rate();

  // Define a TfliteMicroModel instance
  // This allows for executing the compiled model.
  // NOTE: This effectively wraps the Tensorflow-Lite Micro Interpreter.
  TfliteMicroModel model;

  printf("Model Parameters example starting ...\n\n");

  // Dump the compile-time parameters
  print_precompiled_defines();

  // Load the compiled .tflite model
  // with the given TfliteMicroModel instance
  printf("Loading model\n");
  if(!load_tflite_micro_model(model)) {
    printf("Failed to load model\n");
    for(;;);
  }

  printf("The following parameters are accessed at runtime:\n");
  // Print a summary of all the parameters
  // NOTE: This API requires the logging utility.
  model.parameters.print_summary();


  printf("The following demonstrates the TfliteModelParameters API:\n");
  auto& params = model.parameters;

  printf("%d model parameters available:\n", params.size());

  // This demonstrates the iterator
  for(auto e : params)
  {
    printf("%s\n", e->key());
  }

  // This demonstrates the "contains()" API
  if(params.contains("sample_rate"))
  {
    // This demonstrates the [] operator
    auto value = params["sample_rate"];
    printf("Parameters contains entry: sample_rate=%d (type=%d)\n", value->i32(), value->type());
  }

  // This demontrates retrieving a binary array
  const uint8_t* blob_data;
  uint32_t blob_len;
  if(params.get("blob", blob_data, blob_len))
  {
    printf("\"blob\" len=%d\n", blob_len);
  }

  // This demontrates retrieving a string list
  StringList class_labels;
  if(params.get("class_labels", class_labels))
  {
    printf("Class labels=");
    for(auto e : class_labels)
    {
      printf("%s, ", e);
    }
    printf("\n");
  }

  // This demontrates retrieving a float list
  FloatList float_list;
  if(params.get("float_list", float_list))
  {
    printf("Float list=");
    for(auto e : float_list)
    {
      printf("%3.1f, ", e);
    }
    printf("\n");
  }

  // This demontrates retrieving an integer list
  Int32List int_list;
  if(params.get("int_list", int_list))
  {
    printf("Int list=");
    for(auto e : int_list)
    {
      printf("%d, ", e);
    }
    printf("\n");
  }

  // This demontrates retrieving a string
  const char* msg;
  if(params.get("msg", msg))
  {
    printf("msg=%s\n", msg);
  }

  // This demontrates retrieving a bool
  bool normalize_samples;
  if(params.get("normalize_samples", normalize_samples))
  {
    printf("normalize_samples=%d\n", normalize_samples);
  }

  // This demontrates retrieving a float
  float period_ms;
  if(params.get("period_ms", period_ms))
  {
    printf("period_ms=%f\n", period_ms);
  }

  // This demontrates retrieving an int
  int sample_rate;
  if(params.get("sample_rate", sample_rate))
  {
    printf("sample_rate=%d\n", sample_rate);
  }

  printf("Done\n");

  return 0;
}
