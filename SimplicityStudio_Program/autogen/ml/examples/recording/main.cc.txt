/**
 * This demonstrates how to execute a compiled .tflite model
 * with the recording feature enabled.
 *
 * Refer to the corresponding README for more details.
 */

// Include the SDK headers
#include "sl_system_init.h"

// This is an optional header
// that demonstrates how to manually configure the system clocks
#include "ml/platform/ml_clock_helper.h"

// Include your compiled model
// Change "image_classification" to your model's name (if necessary)
#include "keyword_spotting_generated.h"

// To keep things simple,
// we globally use the "npu_toolkit" C++ namespace.
// This way, we do not have to manually use it
// for every call.
using namespace npu_toolkit;


void dump_recorded_data(TfliteMicroModel &model);



int main()
{
  // Initialize the system
  sl_system_init();

  // This is optional,
  // but may be used to configure the platform's max clock rate
  ml_configure_clocks_to_max_rate();

  // Define a TfliteMicroModel instance
  // This allows for executing the compiled model.
  // NOTE: This effectively wraps the Tensorflow-Lite Micro Interpreter.
  TfliteMicroModel model;

  // Obtain a reference to the logger used by the TfliteMicroModel lib
  auto& logger = get_logger();

  // Set the logging level to "debug"
  logger.level(logging::Debug);

  logger.info("Recording example starting ...");

  // Enable the recorder BEFORE loading the model
  model.set_recording_flags(TfliteMicroModel::RecordingFlags::Basic);

  // Load the compiled .tflite model
  // with the given TfliteMicroModel instance
  logger.info("Loading model");
  if(!load_tflite_micro_model(model)) {
    logger.error("Failed to load model");
    for(;;);
  }
  auto acc = model.accelerator();
  logger.info("Using accelerator: %s, %s", acc->name(), acc->description());

  // Dump any recorded data generated while loading the model
  dump_recorded_data(model);


  // Populate the model input tensor
  // Typically, this is when you would write
  // your sensor data (e.g. image, audio, accelerometer, etc.)
  // For this example, we just set them to 0
  for(unsigned i = 0; i < model.n_inputs(); ++i)
  {
    auto& input_tensor = *model.input(i);
    memset(input_tensor.data.raw, 0, input_tensor.bytes);
  }


  // Invoke the model
  // This will cause the ML model to execute on the accelerator
  // The model's output tensors will contain the results
  if(!model.invoke()) {
    logger.error("Failed to invoke model");
    for(;;);
  }

  // For this example, we just print a summary of the output tensor(s)
  logger.info("Output tensor(s):");
  for(unsigned i = 0; i < model.n_outputs(); ++i)
  {
    auto& output_tensor = *model.output(i);
    logger.info("  [%d] %s\n", i, output_tensor.to_str());
  }

  // Dump any recorded data generated while invoking the model
  dump_recorded_data(model);

  logger.info("Done");

  return 0;
}


void dump_recorded_data(
    TfliteMicroModel &model
)
{
    const uint8_t* buffer;
    uint32_t buffer_length;
    msgpack_object_t* root_obj;
    auto& logger = get_logger();

    if(!model.get_recorded_data(&buffer, &buffer_length))
    {
        logger.error("No recorded data available");
        return;
    }
    if(msgpack_deserialize_with_buffer(
        &root_obj,
        buffer,
        buffer_length,
        MSGPACK_FLAGS_NONE
    ) != 0)
    {
        logger.error("Failed to de-serialize recorded data");
        return;
    }

    auto saved_flags = logger.flags(logging::None);

    if(msgpack_dump(root_obj, 10, [](const char* s, void *arg)
    {
        auto& l = *reinterpret_cast<logging::Logger*>(arg);
        l.write_buffer(logging::Info, s);
    }, &logger) != 0)
    {
        logger.error("Error while dumping recorded data");
    }

    logger.flags(saved_flags);

    msgpack_free_objects(root_obj);
}
